1. Error Absoluto

Representa la magnitud de la diferencia entre el valor que medimos o calculamos y el valor verdadero de la magnitud:

Ea = | Valor Real − Valor Aproximado |

Nos dice cuánto nos desviamos, pero no indica qué tan grave es el error en relación al tamaño del número.

#include

using namespace std;

// Calcula error absoluto
double calcularEa(double real, double aprox){
    return fabs(real - aprox);
}

int main(){
    double vr, va;
    int precision;
    cout << "=== CALCULO DE ERROR ABSOLUTO ===" << endl;
    cout << "Ingrese valor real: "; cin >> vr;
    cout << "Ingrese valor aproximado: "; cin >> va;
    cout << "Cifras significativas (2-5): "; cin >> precision;
    
    double resultado = calcularEa(vr, va);
    cout << fixed << setprecision(precision);
    cout << "\nResultado: Ea = " << resultado << endl;
    return 0;
}

2. Error Relativo

Normaliza el error dividiéndolo entre el valor real, permitiendo comparar errores en mediciones de distinta escala:

Er = ( Ea / |Vr| ) × 100 %

Expresado en porcentaje, nos dice qué proporción del valor real representa nuestro error.

// Código C++ Error Relativo
#include 
#include 
using namespace std;
int main(){
    double vr=100, va=97.5;
    double ea = fabs(vr-va);
    double er = (ea / fabs(vr)) * 100;
    cout << "Error Relativo = " << er << " %" << endl;
    return 0;
}

3. Método de Punto Fijo

Reescribimos la ecuación f(x)=0 en la forma equivalente x = g(x). Iteramos hasta que la sucesión converge a la raíz:

xn+1 = g(xn)

Converge si en el intervalo se cumple que |g'(x)| < 1.

// C++ Punto Fijo
#include 
#include 
using namespace std;
double g(double x){ return (5*x - 1) / (2*x*x); } // g(x) para 2x³-5x+1=0
int main(){
    double x=0.5, tol=0.0001, xn;
    for(int i=1; i<=100; i++){
        xn = g(x);
        if(fabs(xn-x) < tol) break;
        x = xn;
    }
    cout << "Raiz: " << xn << endl;
}
